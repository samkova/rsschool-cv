package main
import (
	"fmt"
	"math/rand"
)
func main() {
	x := randomArray(10)
	fmt.Println(x)
	multiplication := 1
	for i := 0; i < len(x); i++ {
		multiplication = multiplication * x[i]
	}
	fmt.Print("Произведение всех элементов ", multiplication)
}
func randomArray(length int) []int {
	result := make([]int, length)
	for i := 0; i < length; i++ {
		result[i] = rand.Intn(100)
	}
	return result
}
